Solution

# Complete the kangaroo function below.
def kangaroo(x1, v1, x2, v2):
    if (x1-x2<0) and (v1-v2<0):
        return "NO"
    k = (x2-x1)/(v2-v1)
    if k.is_integer():
        return "YES"
    else: 
        return "NO"
